// Setting up the map var map = L.map('map').setView([#{venueSearchBean.venue.latitude}, #{venueSearchBean.venue.longitude}], 13); // create the tile layer with correct attribution var osmUrl='http://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png'; var osmAttrib='Map data © <a href="http://openstreetmap.org">OpenStreetMap</a> contributors'; L.tileLayer(osmUrl, {maxZoom: 18, attribution: osmAttrib}).addTo(map); var marker = L.marker([#{venueSearchBean.venue.latitude}, #{venueSearchBean.venue.longitude}]).addTo(map); marker.bindPopup("#{venueSearchBean.venue.name}").openPopup();
